local function walkOnWater()
    local player = game.Players.LocalPlayer
    local character = player.Character or player.CharacterAdded:Wait()
    local humanoid = character:WaitForChild("Humanoid")
    local waterPart = Instance.new("Part")
    waterPart.Size = Vector3.new(1000, 0.1, 1000)
    waterPart.Anchored = true
    waterPart.Transparency = 1
    waterPart.Position = Vector3.new(0, -8, 0)
    waterPart.Parent = game.Workspace
    
    local function updatePosition()
        if character and character:FindFirstChild("HumanoidRootPart") then
            local rootPos = character.HumanoidRootPart.Position
            waterPart.Position = Vector3.new(rootPos.X, waterPart.Position.Y, rootPos.Z)
        end
    end
    
    game:GetService("RunService").Stepped:Connect(updatePosition)
end
